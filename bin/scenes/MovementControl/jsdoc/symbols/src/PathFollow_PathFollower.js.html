 82 realXtend.PathFollow.PathFollower.prototype.ServerInitialize = function()
 83 {
 84 	this.actualpath=0;
 85 	this.direction=0;
 86 	this.nextWayPoint = 0;
 87 	this.nextWayPoint_entity=0;
 88 	this.paused=false;
 89 	this.primera=true;
 90 	this.PathProperties = me.GetOrCreateComponentRaw("EC_DynamicComponent", "PathProperties");
 91 	
 92 	if(!this.PathProperties.ContainsAttribute("Path"))
 93 	{
 94 		/**
 95 		 * [Dynamic] Determines path to follow
 96 		 * @name realXtend.PathFollow.PathFollower#Path
 97 		 * @type int
 98 		 */
 99 		this.PathProperties.CreateAttribute("int", "Path");
100 		
101 		this.PathProperties.SetAttribute("Path", 0);
102 	}	
103 	
104 	if(!this.PathProperties.ContainsAttribute("Direction"))
105 	{
106 		/**
107 		 * [Dynamic] Determines direction the path is followed
108 		 * @name realXtend.PathFollow.PathFollower#Direction
109 		 * @type int
110 		 */
111 		this.PathProperties.CreateAttribute("int", "Direction");
112 				
113 		this.PathProperties.SetAttribute("Direction", 1);
114 	}		
115 	/**
116 	 * [Action] pauseFollow action pauses following.
117 	 * @name realXtend.PathFollow.PathFollower#pauseFollow
118 	 * @event
119 	 */
120 	me.Action("pauseFollow").Triggered.connect(this, this.ServerPauseFollow);
121 	
122 	/**
123 	 * [Action] resumeFollow action resumes following.
124 	 * @name realXtend.PathFollow.PathFollower#resumeFollow
125 	 * @event
126 	 */
127 	me.Action("resumeFollow").Triggered.connect(this, this.ServerResumeFollow);
128 	
129 	/**
130 	 * [Action] followPath starts following a path.
131 	 * @name realXtend.PathFollow.PathFollower#followPath
132 	 * @event
133 	 */
134 	me.Action("followPath").Triggered.connect(this, this.ServerFollowPath);
135 	this.Riggid = me.GetComponentRaw("EC_RigidBody");
136 	
137 	if(this.Riggid){
138 		if(this.Riggid.physicsType==1){
139 			this.Riggid.physicsType=2;
140 		}
141 	}
142 	else{
143 		this.Riggid = me.GetOrCreateComponentRaw("EC_RigidBody");
144 		this.Riggid.phantom=true;
145 		var sizeVec = new Vector3df();
146 		sizeVec.x = 1;
147 		sizeVec.y = 1;
148 		sizeVec.z = 1;
149 		this.Riggid.size=sizeVec;	
150 		this.Riggid.physicsType=2;
151 	}
152 };

155 /**
156  * Start the Path Following if the parameters (Path,Direction) exists.
157  */
158 realXtend.PathFollow.PathFollower.prototype.ServerFollowPath = function() {
159 	this.nextWayPoint = 0;
160 	var pathnumber=this.PathProperties.GetAttribute("Path");
161 	this.actualpath=scene.GetEntityByNameRaw("Path["+pathnumber+"]");
162 	this.direction=this.PathProperties.GetAttribute("Direction");
163 	if(this.actualpath && (this.direction == 1 || this.direction == -1)){
164 			//print("Cambio al Camino: Path["+pathnumber+"]");
165 			if(this.primera){
166 				this.primera=false;
167 				this.Point();
168 			}
169 			else if(!this.primera){
170 				try {
171 					this.nextWayPoint_entity.volumetrigger.EntityEnter.disconnect(this, this.Arrive);
172 				} catch (e) {
173 					print("[PathFollower] Error when disconnecting from volume trigger:" + e.message);
174 					}
175 				me.Exec(2, "Move", 0,0,0);
176 				this.Point();
177 			}
178 		}
179 	else{
180 		print("Parametros incorrectos.");
181 	}
182 };
183 
184 /**
185  * @private
186  */
187 realXtend.PathFollow.PathFollower.prototype.Point = function(){
188 	//print("getNextWaypoint("+this.actualpath.id+","+this.nextWayPoint+","+this.direction+");");
189 	var result = realXtend.PathFollow.getNextWaypoint(this.actualpath.id,this.nextWayPoint,this.direction);
190 	if (result[0]==0){
191 		//print("Path finalizado.");
192 		me.Exec(2, "Move", 0,0,0);
193 		this.primera=true;
194 	}
195 	else if (result[0]!=0){
196 		if (this.nextWayPoint==result[0]){
197 			this.Point();
198 		}
199 		else if (this.nextWayPoint!=result[0]){
200 			this.nextWayPoint = result[0];
201 			this.direction = result[1];
202 			this.nextWayPoint_entity=scene.GetEntityRaw(this.nextWayPoint);
203 			
204 			if (typeof(this.nextWayPoint_entity.volumetrigger) == "undefined" || 
205 					this.nextWayPoint_entity.volumetrigger == null)
206 				return;
207 			
208 			if (this.nextWayPoint_entity.volumetrigger.IsPivotInside(me)) {
209 				print("[PathFollower] entity already placed on next waypoint.");
210 				this.Point();
211 			}
212 			else
213 			{
214 				me.Exec(2, "goto", [this.nextWayPoint_entity.placeable.transform.pos.x,this.nextWayPoint_entity.placeable.transform.pos.y,this.nextWayPoint_entity.placeable.transform.pos.z,"true"]);
215 				//print("goto("+this.nextWayPoint_entity.placeable.transform.pos.x+","+this.nextWayPoint_entity.placeable.transform.pos.y+","+this.nextWayPoint_entity.placeable.transform.pos.z+");");
216 				//this.nextWayPoint_entity.rigidbody.drawDebug = true;
217 				this.nextWayPoint_entity.volumetrigger.EntityEnter.connect(this, this.Arrive);
218 			}
219 		}
220 	}
221 };

223 /**
224  * @private
225  * @param {IEntity*} Entity who arrives to one point.
226  */
227 realXtend.PathFollow.PathFollower.prototype.Arrive = function(Entity){
228 	if (me.id == Entity.id){
229 		//this.nextWayPoint_entity.rigidbody.drawDebug = false;
230 		this.nextWayPoint_entity.volumetrigger.EntityEnter.disconnect(this, this.Arrive);
231 		this.Point();
232 	}
233 };

235 /**
236  * Pauses the Path if the Entity is moving.
237  */
238 realXtend.PathFollow.PathFollower.prototype.ServerPauseFollow = function() {
239 	if(this.paused==true){
240 		//print("Already paused");
241 	}
242 	else{
243 		this.paused=true;
244 		//print("Path following paused");
245 		//this.nextWayPoint_entity.rigidbody.drawDebug = false;
246 		this.nextWayPoint_entity.volumetrigger.EntityEnter.disconnect(this, this.Arrive);
247 		me.Exec(2, "Move", 0,0,0);
248 	}
249 };
250 
251 /**
252  * Resumes the Path if following is paused.
253  */
254 realXtend.PathFollow.PathFollower.prototype.ServerResumeFollow = function() {
255 	if(this.paused==true){
256 		//print("Path following resumed");
257 		this.paused=false;
258 		//print("Tienes que volver al punto: "+this.nextWayPoint);
259 		me.Exec(2, "goto", [this.nextWayPoint_entity.placeable.transform.pos.x,this.nextWayPoint_entity.placeable.transform.pos.y,this.nextWayPoint_entity.placeable.transform.pos.z,"true"]);
260 		//print("goto("+this.nextWayPoint_entity.placeable.transform.pos.x+","+this.nextWayPoint_entity.placeable.transform.pos.y+","+this.nextWayPoint_entity.placeable.transform.pos.z+");");
261 		//this.nextWayPoint_entity.rigidbody.drawDebug = true;
262 		this.nextWayPoint_entity.volumetrigger.EntityEnter.connect(this, this.Arrive);
263 	}
264 	else{
265 		//print("Already following the Path");
266 	}
267 };
