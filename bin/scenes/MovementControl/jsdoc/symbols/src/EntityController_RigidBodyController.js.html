 20 /**
 21  * Creates a RigidBodyController object.
 22  * 
 23  * @class Controller of RigidBody
 24  * @augments realXtend.EntityController.EntityController
 25  */
 26 realXtend.EntityController.RigidBodyController = function ()
 27 {
 28 	var isserver = server.IsRunning();
 29 	if (isserver) 
 30 	{
 31 		this.default_fly_vel_factor = 1.5;
 32 		this.default_run_vel_factor = 2.0;
 33 		this.timeOnGround = 0;
 34 		this.minTimeOnGround = 1;
 35 		this.jump_velocity = 7.0;
 36 		this.fly_jump_meters = 3.0;
 37 		this.angularDamping = 0.5;
 38 	    this.ServerInitialize();
 39 	}
 40 	else
 41 		this.ClientInitialize();
 42 	
 43 };

 59 //COMMON METHODS
 60 /**
 61  * Common update method
 62  * @param {Number} frametime
 63  */
 64 realXtend.EntityController.RigidBodyController.prototype.CommonUpdate = function(frametime)
 65 {
 66 	if (me.rigidbody)
 67 	{
 68 		var velocity = me.rigidbody.linearVelocity;
 69 		velocity = me.placeable.GetAbsoluteVector(velocity);
 70 		var flying = false;
 71 		var onGround = true;
 72 		var running = false;
 73 		
 74 		if (this.syncProperties)
 75 		{
 76 			flying = this.syncProperties.GetAttribute("flying");
 77 			running = this.syncProperties.GetAttribute("running");
 78 			onGround = this.syncProperties.GetAttribute("onGround");
 79 		}
 80 		
 81 		if (onGround) {
 82 			if (this.getMove().z < 0)
 83 				velocity.z = -1;
 84 			else
 85 				velocity.z = 0;
 86 		}
 87 	
 88 		this.SyncObjects(frametime, [velocity, flying, running, onGround]);
 89 	}
 90 	
 91 	if (this.getRotateAutoStop() == true)
 92 	{
 93 		var tmpVec = new Vector3df();
 94 		tmpVec.x = 0;
 95 		tmpVec.y = 0;
 96 		tmpVec.z = 0;
 97 		this.setRotate(tmpVec);
 98 		this.setRotateAutoStop(false);
 99 	}
100 
101 };

180 /**
181  * @see realXtend.EntityController.EntityController#ServerInitialize
182  */
183 realXtend.EntityController.RigidBodyController.prototype.ServerInitialize = function()
184 {
185 	me.GetOrCreateComponentRaw("EC_RigidBody");
186 	
187 	//init sync properties
188 	/**
189 	 * [Dynamic] Determines if entity is flying
190 	 * @name realXtend.EntityController.RigidBodyController#flying
191 	 * @type bool
192 	 */
193 	this.syncProperties.CreateAttribute("bool", "flying");
194 	
195 	/**
196 	 * [Dynamic] Determines if entity flying movement velocity factor
197 	 * @name realXtend.EntityController.RigidBodyController#flyingMovementVelocityFactor
198 	 * @type real
199 	 */
200 	this.syncProperties.CreateAttribute("real", "flyingMovementVelocityFactor");
201 	
202 	/**
203 	 * [Dynamic] Determines if entity is running
204 	 * @name realXtend.EntityController.RigidBodyController#running
205 	 * @type bool
206 	 */
207 	this.syncProperties.CreateAttribute("bool", "running");
208 	
209 	/**
210 	 * [Dynamic] Determines entity running movement velocity factor
211 	 * @name realXtend.EntityController.RigidBodyController#runningMovementVelocityFactor
212 	 * @type real
213 	 */
214 	this.syncProperties.CreateAttribute("real", "runningMovementVelocityFactor");
215 	
216 	/**
217 	 * [Dynamic] Determines if entity is on ground
218 	 * @name realXtend.EntityController.RigidBodyController#onGround
219 	 * @type bool
220 	 */
221 	this.syncProperties.CreateAttribute("bool", "onGround");
222 	
223 	this.syncProperties.SetAttribute("flying", false);
224 	this.syncProperties.SetAttribute("flyingMovementVelocityFactor", this.default_fly_vel_factor);
225 	this.syncProperties.SetAttribute("running", false);
226 	this.syncProperties.SetAttribute("runningMovementVelocityFactor", this.default_run_vel_factor);
227 	this.syncProperties.SetAttribute("onGround", false);	
228 
229 	this.syncProperties.AttributeChanged.connect(this, this.ServerPropertiesChanged);
230 	
231 	// Hook to physics update
232 	me.rigidbody.GetPhysicsWorld().Updated.connect(this, this.ServerUpdate);
233 };

264 /**
265  * Server update method called at every physics simulation tick
266  * @param {Number} frametime
267  */
268 realXtend.EntityController.RigidBodyController.prototype.ServerUpdate = function(frametime)
269 {   
270 	var onGround = this.syncProperties.GetAttribute("onGround");
271 	var flying = this.syncProperties.GetAttribute("flying");
272 	var goToLooking = 0;
273 	
274 	if (!onGround && !flying)
275 	{
276 		//calculate if rigid body is onGround	
277 		if (this.IsOnGround(frametime))
278 		{
279 			this.syncProperties.SetAttribute("onGround",true);
280 			onGround = true;
281 		}
282 	}
283 	
284 	//if is flying and is onground. goTo x meters up
285 	if (flying && onGround)
286 	{
287 		var tmpVec = new Vector3df();
288 		tmpVec.x = me.placeable.transform.pos.x;
289 		tmpVec.y = me.placeable.transform.pos.y;
290 		tmpVec.z = me.placeable.transform.pos.z + this.fly_jump_meters;
291 		this.setGoTo(tmpVec);
292 		this.syncProperties.SetAttribute("onGround", false);
293 	}
294 	
295 	//move rigidbody
296 	var moveVec = new Vector3df();
297 	if ((this.getMove().x != 0) || (this.getMove().y != 0) || (this.getMove().z != 0))
298 	{
299 		this.ResetAutoUpdatedVariables();
300 		moveVec = this.getMove();
301 		moveVec = me.placeable.GetRelativeVector(moveVec);
302 	}
303 	else if (this.getGoTo() != 0)
304 	{
305 		//calc lookat vector and set this.lookAt
306 		var placeable_pos = me.placeable.transform.pos;
307 		var dest = new Vector3df();
308 		dest.x = this.getGoTo().x;
309 		dest.y = this.getGoTo().y;
310 		dest.z = this.getGoTo().z;
311 		
312 		//if rigid body is not flying and is on ground move rigid body in xy axis
313 		if (!flying && onGround)
314 		{
315 			placeable_pos.z = 0;
316 			dest.z = 0;
317 		}
318 		
319 		if ((Math.abs(placeable_pos.x - dest.x) < this.goToThreshold) && (Math.abs(placeable_pos.y - dest.y) < this.goToThreshold) && (Math.abs(placeable_pos.z - dest.z) < this.goToThreshold))
320 		{
321 			this.StopMove(true);
322 		}
323 		else
324 		{
325 			//set move vector 			
326 			moveVec.x = dest.x - placeable_pos.x;
327 			moveVec.y = dest.y - placeable_pos.y;
328 			moveVec.z = dest.z - placeable_pos.z;
329 			
330 			moveVec = moveVec.normalize();
331 			//print("goTo moveVec = x:" + moveVec.x + " y:" + moveVec.y + " z:" + moveVec.z);
332 			
333 			//set lookat if goTolooking is true
334 			if (this.getGoToLooking() == true)
335 				goToLooking = this.getGoTo();
336 		}
337 	}
338 	//if there is movement
339 	if ((moveVec.x != 0) || (moveVec.y != 0) || (moveVec.z != 0) || me.rigidbody.IsActive())
340 	{		
341 		this.Move(moveVec.x, moveVec.y, moveVec.z, frametime);
342 	}
343 	
344 	//rotate
345 	var rotateVec = new Vector3df();
346 	if ((this.getRotate().x != 0) || (this.getRotate().y != 0) || (this.getRotate().z != 0))
347 	{
348 		this.ResetAutoUpdatedVariables();
349 		rotateVec = this.getRotate();
350 	}
351 	else if ((this.getLookAt() != 0) || (this.getRotateTo() != 0) || (goToLooking != 0))
352 	{
353 		var lookAtRot = 0;
354 		var lookAt = 0;
355 		if (goToLooking != 0) //first check if goTo looking is enabled
356 			lookAt = goToLooking;
357 		else if (this.getLookAt() != 0)
358 			lookAt = this.getLookAt();
359 
360 		if (lookAt) //set lookAtRot for lookat variable
361 		{			
362 			var rotationMaskX = this.syncProperties.GetAttribute("rotationMaskX");
363 			var rotationMaskY = this.syncProperties.GetAttribute("rotationMaskY");
364 					
365 			var diff = new Vector3df();
366 			var current_pos = me.placeable.transform.pos;
367 			diff.x = lookAt.x - current_pos.x;
368 			diff.y = lookAt.y - current_pos.y;
369 			diff.z = lookAt.z - current_pos.z;		
370 			if ((rotationMaskX==0) && (rotationMaskY==0))
371 				diff.z = 0;
372 			
373 			var mov_dir = new Vector3df();
374 			mov_dir.x = 1;
375 			
376 			mov_dir = me.placeable.GetRelativeVector(mov_dir);
377 			
378 			lookAtRot = me.placeable.GetRotationFromTo(mov_dir,diff);			
379 		}
380 		else if (this.getRotateTo() != 0)  //set lookAtRot for rotateTo variable
381 		{
382 			lookAtRot = new Vector3df();			
383 			lookAtRot.x	=  this.getRotateTo().x - me.placeable.transform.rot.x;
384 			lookAtRot.y =  this.getRotateTo().y - me.placeable.transform.rot.y;
385 			lookAtRot.z =  this.getRotateTo().z - me.placeable.transform.rot.z;					
386 		}
387 		
388 		if (lookAtRot != 0)
389 		{
390 			lookAtRot = this.NormalizeRotation(lookAtRot);
391 			//print(lookAtRot.x + " "+ lookAtRot.y +" "+lookAtRot.z);
392 			//rotate with default velocity (set 1 or -1 rotateVec)
393 			if (lookAtRot.z > this.lookAtThreshold)
394 				rotateVec.z = 1;
395 			else if (lookAtRot.z < -this.lookAtThreshold)
396 				rotateVec.z = -1;
397 			if (lookAtRot.y > this.lookAtThreshold)
398 				rotateVec.y = 1;
399 			else if (lookAtRot.y < -this.lookAtThreshold)
400 				rotateVec.y = -1;			
401 			if (lookAtRot.x > this.lookAtThreshold)
402 				rotateVec.x = 1;
403 			else if (lookAtRot.x < -this.lookAtThreshold)
404 				rotateVec.x = -1;
405 
406 			
407 			if ((rotateVec.x == 0) && (rotateVec.y == 0) && (rotateVec.z == 0))			
408 				this.StopRotate(true);		
409 		}
410 	}			
411 	
412 	//rotate rigidbody
413 	this.Rotate(rotateVec.x, rotateVec.y, rotateVec.z);	
414 	
415 	this.CommonUpdate(frametime);
416 };

418 /**
419  * Checks if rigid body is on ground or not
420  * @param {Number} frametime
421  * @returns {Boolean}
422  */
423 realXtend.EntityController.RigidBodyController.prototype.IsOnGround = function(frametime) 
424 {
425 	if (Math.abs(me.rigidbody.linearVelocity.z) < 0.005)
426 		this.timeOnGround += frametime;
427 	if (this.timeOnGround != 0 && Math.abs(me.rigidbody.linearVelocity.z) > 0.005)
428 		this.timeOnGround = 0;
429 	if (this.timeOnGround > this.minTimeOnGround)
430 	{	
431 		this.timeOnGround = 0;
432 		return true;
433 	}
434 	return false;
435 };

437 /**
438  * Performs rotation
439  * @param {Number} x
440  * @param {Number} y
441  * @param {Number} z
442  */
443 realXtend.EntityController.RigidBodyController.prototype.Rotate = function(x, y, z) 
444 {
445 	//rotate rigid body
446 	if ((x != 0) || (y != 0) || (z != 0) || me.rigidbody.IsActive())
447 	{
448 		var velocity = this.syncProperties.GetAttribute("rotationVelocity");
449 		if (!velocity)
450 			return;
451 		
452 		var rotationMaskX = this.syncProperties.GetAttribute("rotationMaskX");
453 		var rotationMaskY = this.syncProperties.GetAttribute("rotationMaskY");
454 		var rotationMaskZ = this.syncProperties.GetAttribute("rotationMaskZ");
455 		
456 		if (isNaN(rotationMaskX) || isNaN(rotationMaskY) || isNaN(rotationMaskZ))
457 			return;
458 		
459 		
460         var rotateVec = new Vector3df();
461         rotateVec.x = x *rotationMaskX;
462         rotateVec.y = y *rotationMaskY;
463         rotateVec.z = z *rotationMaskZ;        
464         
465         var v = new Vector3df();
466         v.x = velocity * rotateVec.x/180 * Math.PI;
467         v.y = velocity * rotateVec.y/180 * Math.PI;
468         v.z = velocity * rotateVec.z/180 * Math.PI;
469         
470 		//transform velocity vector to placeable coordinates
471 		//TorqueImpulse: a * t = ((vf - vi)/ t) * t = (vf - vi)
472 		var v_current = me.rigidbody.angularVelocity;
473 		//print("v_g = z:" + v.z);
474 		//print("v_c = z:" + v_current.z);
475 		rotateVec.x = (v.x - v_current.x/180*Math.PI);
476 		rotateVec.y = (v.y - v_current.y/180*Math.PI);
477 		rotateVec.z = (v.z - v_current.z/180*Math.PI);
478 		
479 		//Apply damping. 
480 		if (x == 0)
481 			rotateVec.x *= this.angularDamping;
482 		if (y == 0)
483 			rotateVec.y *= this.angularDamping;
484 		if (z == 0)
485 			rotateVec.z *= this.angularDamping;
486 		//print("Rotating rigid body: x=", rotateVec.x, " y=", rotateVec.y, " z=", rotateVec.z);
487 		me.rigidbody.ApplyTorqueImpulse(rotateVec);
488     }
489 };

491 /**
492  * Performs movement
493  * @param {Number} x
494  * @param {Number} y
495  * @param {Number} z
496  * @param {Number} frametime
497  */
498 realXtend.EntityController.RigidBodyController.prototype.Move = function(x, y, z, frametime)
499 {
500 	var moveVec = new Vector3df();
501 	moveVec.x = x;
502 	moveVec.y = y;
503 	moveVec.z = z;
504 	
505 	var onGround = this.syncProperties.GetAttribute("onGround");
506 	var flying = this.syncProperties.GetAttribute("flying");
507 	var running = this.syncProperties.GetAttribute("running");
508 	
509 	var jump = false;
510 	var velocity = this.syncProperties.GetAttribute("movementVelocity");		
511 	if (flying)
512 		velocity *= this.syncProperties.GetAttribute("flyingMovementVelocityFactor");
513 	else if (running)
514 		velocity *= this.syncProperties.GetAttribute("runningMovementVelocityFactor");
515 	
516     // Apply motion force
517    
518 	var impulseVec = new Vector3df();
519 	// velocity
520 	var v = new Vector3df();
521 	v.x = velocity*moveVec.x;
522 	v.y = velocity*moveVec.y;
523 	v.z = velocity*moveVec.z;
524 	
525 	if (this.getMove().z > 0 && onGround) {
526 		// jump velocity
527 		v.z = this.jump_velocity;
528 		this.getMove().z = 0;
529 		this.syncProperties.SetAttribute("onGround", false);
530 		jump = true;
531 	}
532 	//transform velocity vector to placeable coordinates
533 	//var v_g = me.placeable.GetRelativeVector(v);
534 	//Impulse: F * t 
535 	//Force: m * a 
536 	//Acceleration = ((vf - vi)/ t)
537 	//Impulse = (vf - vi) * m
538 	var v_current = me.rigidbody.linearVelocity;			
539 	impulseVec.x = (v.x - v_current.x) * me.rigidbody.mass;		
540 	impulseVec.y = (v.y - v_current.y) * me.rigidbody.mass;
541 	if (flying || jump)
542 		impulseVec.z = (v.z - v_current.z) * me.rigidbody.mass;
543 		
544 	
545 	if (Math.abs(impulseVec.x) > 0.5 || Math.abs(impulseVec.y) > 0.5 || Math.abs(impulseVec.z) > 0.5)
546 	{
547 		me.rigidbody.ApplyImpulse(impulseVec);
548 		//print("v = x:" + v_current.x + " y:" + v_current.y + " z:" + v_g.z)
549 		//print("impulse = x:" + impulseVec.x + " y:" + impulseVec.y + " z:" + impulseVec.z);
550 	}	
551 };

553 /**
554  * Handles {@link realXtend.EntityController.RigidBodyController#flying} changes
555  */
556 realXtend.EntityController.RigidBodyController.prototype.ServerHandleFly = function() {
557     var rigidbody = me.rigidbody;
558 
559     var flying = this.syncProperties.GetAttribute("flying");
560     if (flying) {
561 		var gravity = new Vector3df(0,0,0);
562 		rigidbody.ApplyGravity(gravity);		
563     } else {
564 		rigidbody.ApplyGravity(rigidbody.GetPhysicsWorld().GetGravity());
565         // Reset the x rot if left
566         var av_placeable = me.placeable;
567         var av_transform = av_placeable.transform;
568         if (av_transform.rot.x != 0) {
569             av_transform.rot.x = 0;
570             av_placeable.transform = av_transform;
571         }
572         // Push rigid body a bit to the fly direction
573         // so the motion does not just stop to a wall
574         var moveVec = new Vector3df();
575         moveVec.x = this.getMove().x * 120;
576         moveVec.y = this.getMove().y * 120;
577         moveVec.z = this.getMove().z * 120;
578         var pushVec = av_placeable.GetRelativeVector(moveVec);
579         me.rigidbody.ApplyImpulse(pushVec);		
580     }
581 };
